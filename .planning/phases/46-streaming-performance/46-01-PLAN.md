    ```typescript
    describe("PERF-01: rAF throttle coalescing", () => {
      beforeEach(() => { vi.useFakeTimers() })
      afterEach(() => { vi.useRealTimers() })

      it("scheduleUpdate guard: multiple rapid calls result in one pushUpdate", () => {
        // Replicate the closure logic inline for unit testing
        let updateScheduled = false
        let updateTimer: ReturnType<typeof setTimeout> | null = null
        let pushUpdateCallCount = 0
        const pushUpdate = () => { pushUpdateCallCount++ }

        function scheduleUpdate() {
          if (updateScheduled) return
          updateScheduled = true
          updateTimer = setTimeout(() => {
            updateTimer = null
            requestAnimationFrame(() => {
              updateScheduled = false
              pushUpdate()
            })
          }, 50) // STREAM_THROTTLE_MS
        }

        // Simulate 10 rapid deltas
        for (let i = 0; i < 10; i++) scheduleUpdate()

        // Advance past throttle window
        vi.advanceTimersByTime(100)
        // Flush rAF (jsdom: requestAnimationFrame fires on next timer tick in fake timers)
        vi.runAllTimers()

        expect(pushUpdateCallCount).toBe(1) // 10 calls → 1 pushUpdate
      })

      it("scheduleUpdate: second call within window is a no-op (no duplicate setTimeout)", () => {
        const setTimeoutSpy = vi.spyOn(global, 'setTimeout')

        let updateScheduled = false
        function scheduleUpdate() {
          if (updateScheduled) return
          updateScheduled = true
          setTimeout(() => { updateScheduled = false }, 50)
        }

        scheduleUpdate()
        scheduleUpdate() // should be no-op
        scheduleUpdate() // should be no-op

        expect(setTimeoutSpy).toHaveBeenCalledTimes(1)
        setTimeoutSpy.mockRestore()
      })
    })
    ```
