  <action>
    **Step 1: Define error type and classifier**

    In ChatThread.tsx, above the ErrorBanner function, add:
    ```typescript
    type StreamErrorType = "connection_lost" | "timeout" | "api_error";

    export function classifyStreamError(error: string): StreamErrorType {
      const lower = error.toLowerCase();
      if (lower.includes("connection lost") || lower.includes("failed to fetch") ||
          lower.includes("network") || lower.includes("disconnected") ||
          lower.includes("aborted")) {
        return "connection_lost";
      }
      if (lower.includes("timeout") || lower.includes("timed out")) {
        return "timeout";
      }
      return "api_error";
    }
    ```
    Export classifyStreamError so tests can import it.
